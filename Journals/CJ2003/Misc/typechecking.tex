Let us recall that \ASN subtyping is powerful enough to imply the
typechecking relationship. 

\begin{center}
\begin{inferencesystem}
%
% Type reference
%
\deduction
%%   x \in {\cal A}
%%   \cr
   \Gamma(x) \lhd (\normT', \sigma')
   \cr
   \TypeEnv \vdash v : \normT'
   ----------------------------------
   \TypeEnv \vdash v : \TRef \, (x)
\end
\neardeductionlabel{\hspace*{-4pt}\rulename{TRef}}
\end{inferencesystem}
\begin{inferencesystem}
%
% CHOICE
%
\deduction
  \cr
  \TypeEnv \vdash v' : \FieldEnv{}(l')
 ------------------------------------------------------------------
  \TypeEnv \vdash (l' \colon v') : \kwdCHOICE \, \FieldEnv{}
\end
\neardeductionlabel{\hspace*{-4pt}\rulename{Choice}}
\end{inferencesystem}
\end{center}

\begin{center}
\begin{inferencesystem}
%%====
%% SET
%%====
%%
%% Matched field
%%
\deduction
   \CompEnv{}(l) \lhd (\normT', \wild)
   \cr
   \TypeEnv \vdash \IdEnv{}(l) : \normT'
   \cr
   \TypeEnv \vdash \Map \, (\IdEnv{\Labels' \backslash l}) : \kwdSET \, \CompEnv{\Labels \backslash l}
 ------------------------------------------------------------------------------------
   \TypeEnv \vdash \Map \, (\IdEnv{\Labels'})  : \kwdSET \, \CompEnv{\Labels}
\end
\neardeductionlabel{\hspace*{-4pt}\rulename{Set}}
\end{inferencesystem}
\begin{inferencesystem}
%%
%% Empty SET
%%
\axiom
  \cr \cr
   \TypeEnv \vdash \Map \, (\IdEnv{\varnothing}) : \kwdSET \, \CompEnv{\varnothing}
\end
\neardeductionlabel{\rulename{$\varnothing$-\textsf{set}}}
\end{inferencesystem}
\end{center}

\begin{center}
\begin{inferencesystem}
%%
%% Unmatched OPTIONAL field
%%
\deduction
   \CompEnv{}(l) \lhd (\wild, \Some \, \kwdOPTIONAL)
   \cr
   \TypeEnv \vdash v : \kwdSET \, \CompEnv{\Labels \backslash l}
  ----------------------------------------------------------------
   \TypeEnv \vdash v : \kwdSET \, \CompEnv{\Labels}
\end
\neardeductionlabel{\rulename{OptSet}}
\end{inferencesystem}
\end{center}
